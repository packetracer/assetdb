CREATE DATABASE python_DB;
USE python_DB;
CREATE TABLE HOST (id INT AUTO_INCREMENT PRIMARY KEY, FACID INT, host VARCHAR(255) UNIQUE, MODELID INT, VERSIONID INT);
CREATE TABLE FACILITY (id INT AUTO_INCREMENT PRIMARY KEY, facility VARCHAR(255));
CREATE TABLE MODEL (id INT AUTO_INCREMENT PRIMARY KEY, model VARCHAR(255));
CREATE TABLE VERSION (id INT AUTO_INCREMENT PRIMARY KEY, version VARCHAR(255));
CREATE TABLE SERIAL (id INT AUTO_INCREMENT PRIMARY KEY, serial VARCHAR(255) NOT NULL, HOSTID INT);
